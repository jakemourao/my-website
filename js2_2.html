<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Word Count</title>

</head>

<body>

    <div id="answer"></div>


    <script>

        function countWords() {

            let sentence = prompt("Enter a sentence:");

            if (sentence) {

                let words = sentence.split(" ");

                let wordCount = words.length;

                document.getElementById("answer").textContent = "Number of words: " + wordCount;

            } else {

                document.getElementById("answer").textContent = "No input provided.";

            }

        }


        countWords();

    </script>

</body>

</html>
