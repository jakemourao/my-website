    <h2>Odd Numbers in Range:</h2>

    <p id="odd-numbers"></p>


    <h2>Sum of All Numbers in Range:</h2>

    <p id="sum-result"></p>


    <script>

        // Get the two numbers from the user

        let num1 = prompt("Enter the first number:");

        let num2 = prompt("Enter the second number:");


        // Convert to numbers

        num1 = parseInt(num1);

        num2 = parseInt(num2);


        // Ensure valid input

        if (isNaN(num1) || isNaN(num2)) {

            document.getElementById("odd-numbers").textContent = "Invalid input. Please enter numbers.";

            document.getElementById("sum-result").textContent = "";

        } else {

            // Determine the start and end of the range

            let start = Math.min(num1, num2);

            let end = Math.max(num1, num2);


            let sum = 0;

            let oddNumbers = "";


            // Loop through the range

            for (let i = start; i <= end; i++) {

                sum += i; // Add to sum


                if (i % 2 !== 0) { // Check if odd

                    oddNumbers += i + " ";

                }

            }


            // Display results on the page

            document.getElementById("odd-numbers").textContent = oddNumbers.trim();

            document.getElementById("sum-result").textContent = sum;

        }

    </script>
