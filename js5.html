    <title>Times Table</title>

<body>


    <h2>Multiplication Tables</h2>

    <div id="output"></div>


    <script>

        function timesTable(number) {

            // Get the output div

            let outputDiv = document.getElementById("output");


            // Create a new paragraph element to hold the times table

            let tableContent = document.createElement("p");


            // Add table title

            tableContent.innerHTML = `<h3>Times Table for ${number}</h3>`;


            // Generate the multiplication table up to 12

            for (let i = 1; i <= 12; i++) {

                tableContent.innerHTML += `${number} x ${i} = ${number * i} <br>`;

            }


            // Append the table content to the output div

            outputDiv.appendChild(tableContent);

        }


        // Call timesTable() function with values 3, 7, and 11

        timesTable(3);


        // Insert horizontal rule after the first table

        document.getElementById("output").innerHTML += "<hr>";


        timesTable(7);


        // Insert horizontal rule after the second table

        document.getElementById("output").innerHTML += "<hr>";


        timesTable(11);

    </script>
