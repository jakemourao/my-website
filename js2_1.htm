    <script>

        function count(numbers, value) {

            let count = 0;

            for (let num of numbers) {

                if (num < value) {

                    count++;

                }

            }

            return count;

        }


        function testCount() {

            let numbers = [3, 55, 30, 28, 9, 12];

            let value = 10;

            let result = count(numbers, value);

            document.getElementById("answer").textContent = "Count of numbers less than " + value + ": " + result;

        }


        testCount();

    </script>
